# Include headers
INCLUDE_DIRECTORIES(${PROJECT_SOURCE_DIR}/src)
INCLUDE_DIRECTORIES(${PROJECT_SOURCE_DIR}/tools)

FIND_PACKAGE(GMP REQUIRED)
INCLUDE_DIRECTORIES(${GMP_INCLUDE_DIR})

SET(DEPENDENT_LIBS "")

# Cork
FIND_PACKAGE(Cork REQUIRED)
IF (CORK_FOUND)
    INCLUDE_DIRECTORIES(${CORK_INCLUDE_DIR})
    LIST(APPEND DEPENDENT_LIBS ${CORK_LIBRARIES})
    ADD_DEFINITIONS(-DWITH_CORK)
ENDIF (CORK_FOUND)

# Clipper
FIND_PACKAGE(Clipper REQUIRED)
IF(CLIPPER_FOUND)
    INCLUDE_DIRECTORIES(${CLIPPER_INCLUDE_DIR})
    LIST(APPEND DEPENDENT_LIBS ${CLIPPER_LIBRARIES})
    ADD_DEFINITIONS(-DWITH_CLIPPER)
ENDIF(CLIPPER_FOUND)

# Enumerate source files
FILE(GLOB_RECURSE TEST_SRC_FILES *.cpp)
FILE(GLOB_RECURSE TEST_INC_FILES *.h *.inl)

ADD_EXECUTABLE(csg_tests ${TEST_SRC_FILES} ${TEST_INC_FILES})
TARGET_LINK_LIBRARIES(csg_tests ${GTEST_BOTH_LIBRARIES} Mesh lib_csg
    lib_MeshUtils ${DEPENDENT_LIBS} ${GMP_LIBRARIES})
ADD_CUSTOM_COMMAND(TARGET csg_tests POST_BUILD COMMAND csg_tests)
