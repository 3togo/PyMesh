# Enumerate source files
FILE(GLOB SRC_FILES *.cpp *.inl)
FILE(GLOB INC_FILES *.h)

ADD_LIBRARY(BVHDependencies INTERFACE IMPORTED)

# CGAL
IF (CGAL_FOUND)
    TARGET_LINK_LIBRARIES(BVHDependencies
        INTERFACE
            lib_CGAL
    )
    ADD_SUBDIRECTORY(CGAL)
ENDIF (CGAL_FOUND)

# GeoGram
IF (GEOGRAM_FOUND)
    TARGET_LINK_LIBRARIES(BVHDependencies
        INTERFACE
            lib_Geogram
    )
    ADD_SUBDIRECTORY(Geogram)
ENDIF (GEOGRAM_FOUND)

# IGL
IF (LIBIGL_FOUND)
    TARGET_LINK_LIBRARIES(BVHDependencies
        INTERFACE
            lib_IGL
    )
    ADD_SUBDIRECTORY(IGL)
ENDIF (LIBIGL_FOUND)

ADD_LIBRARY(lib_BVH SHARED ${SRC_FILES} ${INC_FILES})
SET_TARGET_PROPERTIES(lib_BVH PROPERTIES OUTPUT_NAME "PyMesh-BVH")
TARGET_LINK_LIBRARIES(lib_BVH
    PUBLIC
        Mesh
        PyMesh::Tools
    PRIVATE
        BVHDependencies
)
