INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})


# Enumerate source files
FILE(GLOB SRC_FILES *.cpp *.inl)
FILE(GLOB INC_FILES *.h)

SET(DEPENDENT_LIBS "")
SET(DEPENDENT_INCLUDES "")

# Draco
IF (draco_FOUND)
    MESSAGE("Draco found!")
    LIST(APPEND DEPENDENT_INCLUDES ${draco_INCLUDE_DIRS})
    LIST(APPEND DEPENDENT_LIBS ${draco_LIBRARIES})
    ADD_DEFINITIONS(-DWITH_DRACO)
    ADD_SUBDIRECTORY(Draco)
ENDIF (draco_FOUND)

ADD_LIBRARY(lib_compression SHARED ${SRC_FILES} ${INC_FILES})
SET_TARGET_PROPERTIES(lib_compression PROPERTIES OUTPUT_NAME "compression")
TARGET_INCLUDE_DIRECTORIES(lib_compression SYSTEM PRIVATE ${DEPENDENT_INCLUDES})
TARGET_LINK_LIBRARIES(lib_compression Mesh ${DEPENDENT_LIBS})

ADD_CUSTOM_TARGET(compression DEPENDS lib_compression)
