# Include this directory
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})

FIND_PACKAGE(GMP REQUIRED)
INCLUDE_DIRECTORIES(${GMP_INCLUDE_DIR})

IF (DEFINED ENV{CORK_PATH})
    INCLUDE_DIRECTORIES($ENV{CORK_PATH}/include)
    LINK_DIRECTORIES($ENV{CORK_PATH}/lib)
ENDIF (DEFINED ENV{CORK_PATH})

# Enumerate source files
FILE(GLOB SRC_FILES *.cpp *.inl)
FILE(GLOB INC_FILES *.h)

ADD_SUBDIRECTORY(Cork)

ADD_LIBRARY(lib_csg SHARED ${SRC_FILES} ${INC_FILES})
SET_TARGET_PROPERTIES(lib_csg PROPERTIES OUTPUT_NAME "csg")
TARGET_LINK_LIBRARIES(lib_csg Mesh lib_MeshUtils cork ${GMP_LIBRARIES})

# Boudle lib, swig wrapper into a single target.
# Usage: make csg
ADD_CUSTOM_TARGET(csg DEPENDS lib_csg)
